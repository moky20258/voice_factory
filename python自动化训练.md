六、完整 Python 代码（核心）

import os
import json
import random
import requests
from tqdm import tqdm

# =========================
# Fish Speech API 地址
# =========================
API_URL = "http://127.0.0.1:7860/v1/tts"

# =========================
# 输入文本目录
# =========================
TEXT_DIR = "texts"

# =========================
# 输出目录
# =========================
OUTPUT_DIR = "outputs"

# =========================
# metadata目录
# =========================
META_DIR = "metadata"

os.makedirs(OUTPUT_DIR, exist_ok=True)
os.makedirs(META_DIR, exist_ok=True)

# =========================
# 要生成多少个speaker
# =========================
NUM_SPEAKERS = 20

# =========================
# 每个speaker生成多少句
# =========================
LINES_PER_SPEAKER = 30

# =========================
# 读取文本
# =========================
texts = []

for file in os.listdir(TEXT_DIR):
    if file.endswith(".txt"):
        path = os.path.join(TEXT_DIR, file)

        with open(path, "r", encoding="utf-8") as f:
            content = f.read().strip()

            if content:
                texts.append(content)

print(f"加载文本数量: {len(texts)}")

# =========================
# 开始生成
# =========================
for speaker_id in range(1, NUM_SPEAKERS + 1):

    speaker_name = f"speaker_{speaker_id:04d}"

    speaker_dir = os.path.join(OUTPUT_DIR, speaker_name)

    os.makedirs(speaker_dir, exist_ok=True)

    # =========================
    # 随机参数
    # =========================
    seed = random.randint(1000, 999999)

    temperature = round(random.uniform(0.6, 0.9), 2)

    top_p = round(random.uniform(0.7, 0.9), 2)

    repetition_penalty = round(random.uniform(1.1, 1.25), 2)

    # metadata
    metadata = {
        "speaker": speaker_name,
        "seed": seed,
        "temperature": temperature,
        "top_p": top_p,
        "repetition_penalty": repetition_penalty
    }

    with open(
        os.path.join(META_DIR, f"{speaker_name}.json"),
        "w",
        encoding="utf-8"
    ) as f:
        json.dump(metadata, f, indent=4, ensure_ascii=False)

    # =========================
    # 随机抽取句子
    # =========================
    selected_texts = random.sample(
        texts,
        min(LINES_PER_SPEAKER, len(texts))
    )

    print(f"\n开始生成: {speaker_name}")

    for idx, text in enumerate(tqdm(selected_texts)):

        payload = {
            "text": text,
            "temperature": temperature,
            "top_p": top_p,
            "repetition_penalty": repetition_penalty,
            "seed": seed
        }

        try:

            response = requests.post(API_URL, json=payload)

            if response.status_code == 200:

                output_path = os.path.join(
                    speaker_dir,
                    f"{idx:04d}.wav"
                )

                with open(output_path, "wb") as f:
                    f.write(response.content)

            else:

                print(
                    f"生成失败: {response.status_code}"
                )

        except Exception as e:

            print("错误:", e)

print("\n全部生成完成")
